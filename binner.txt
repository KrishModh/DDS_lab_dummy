#include <stdio.h>
#define n 10
int arr[n];
int no;
int BinSearch(int low, int high) {
	int mid;
	if (low > high) {
		return -1;
	}
	mid = (low + high) / 2;
	if (arr[mid] == no) {
		return mid;
	} else if (no > arr[mid]) {
		return BinSearch(mid + 1, high);
	} else {
		return BinSearch(low, mid - 1);
	}
}
int main() {
	int sorted_arr[n] = {2, 5, 8, 12, 16, 23, 38, 56, 72, 91};
	int i;
	for (i = 0; i < n; i++) {
		arr[i] = sorted_arr[i];
	}
	no = 23;
	int index = BinSearch(0, n - 1);
	if (index != -1) {
		printf("Element %d found at index %d.\n", no, index);
	} else {
		printf("Element %d not found in the array.\n", no);
	}
	no = 50;
	index = BinSearch(0, n - 1);
	if (index != -1) {
		printf("Element %d found at index %d.\n", no, index);
	} else {
		printf("Element %d not found in the array.\n", no);
	}
	return 0;
}
